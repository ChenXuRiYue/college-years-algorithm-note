#### 矩阵

- 常见应用：
  - 给定路的长度。查询路径。

- 一个n阶矩阵板子。其中，乘法由快速幂的思想优化。总体上的复杂度是$n^3log(k)$

```cpp
const int maxn = 100 + 10;

const int mod = 1e9 + 7;

struct matrix
{
    static const int n = maxn;
    ll M[n][n];
    int N;
    matrix(int num)
    {
        N = num;
        memset(M, 0, sizeof(M));
    }
    void build()
    {
        for (int i = 1; i <= N; i++)
            M[i][i] = 1;
    }
    ll *operator[](int i)
    {
        return M[i];
    }
    matrix operator*(matrix &a)
    {
        matrix temp(N);
        for (int i = 1; i <= N; i++)
            for (int j = 1; j <= N; j++)
                for (int k = 1; k <= N; k++)
                    temp[i][j] = (temp[i][j] + (1LL * M[i][k] * a[k][j]) % mod) % mod;
        return temp;
    }
    matrix operator^(ll P)
    {
        matrix res(N);
        res.build();
        matrix A = *this;
        while (P > 0)
        {
            if (P & 1)
                res = res * A;
            A = A * A;
            P >>= 1;
        }
        return res;
    }
};
```
