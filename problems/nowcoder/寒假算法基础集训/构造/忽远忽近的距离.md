#include<bits/stdc++.h>
using namespace std;
using ll = long long;

const int N = 1E6 + 10;

void out4(int f)
{
	cout << f * 4 + 3 << ' ' << f * 4 + 4 << ' ' << f * 4 + 1 << ' ' << f * 4 + 2 << ' ';
}

void out5(int l )
{
	l--;
	cout << l + 4 << ' ' << l + 5 << ' ' << l + 1 << ' ' << l + 2 << ' ' << l + 3 << ' ';
}

void out6( int l)
{
	l--;
	cout << l + 4 << ' ' << l + 5 << ' ' << l + 6 << ' ' <<  l + 1 << ' ' << l + 2 << ' ' << l + 3 << ' ';
}




int main()
{
	ios::sync_with_stdio(false);

	int n;

	if (n < 4 || n == 7 )

	if (n % 4 == 1)
	{
		n -= 5;

			out4(i);
